// A: Nro  B: Fecha  C: Llave  D: Monto  E: Foto(URL)  F: Monto Foto
// G: Estado  H: Validado Por  I: Fecha Validación  J: Resp. Envío
// K: Pago Original  L: Tipo Pago Val.  M: Vuelto  N: Motivo Cancelación

function doGet(e) {
  return HtmlService.createHtmlOutput("API Activa - v3.0 Columnas Limpias");
}

function doPost(e) {
  const output = ContentService.createTextOutput().setMimeType(ContentService.MimeType.JSON);
  try {
    const payload = JSON.parse(e.postData.contents);
    const action = payload.action;
    let result = {};
    switch(action) {
      case 'login':               result = login(payload.user, payload.pass); break;
      case 'listarPedidos':       result = listarPedidos(); break;
      case 'crearPedido':         result = crearPedido(payload); break;
      case 'validarPedido':       result = validarPedido(payload); break;
      case 'rechazarPedido':      result = rechazarPedido(payload); break;
      case 'crearPedidosMasivos': result = crearPedidosMasivos(payload); break;
      case 'marcarSLAFuera':      result = marcarSLAFuera(payload.nro, payload.fuera_sla); break;
      case 'guardarFotoOcr':      result = guardarFotoOcrSilent(payload.llave, payload.fotoBase64); break;
      case 'processVoucherOCR':   result = processVoucherOCR(payload); break;
      case 'setGeminiKey':        result = setGeminiKey(payload); break;
      case 'loginMotorizado':     result = loginMotorizado(payload.user, payload.pass); break;
      case 'marcarPorValidar':    result = marcarPorValidar(payload.nro, payload.fechaEntrega, payload.horaEntrega); break;
      case 'asignarMotorizado':   result = asignarMotorizado(payload); break;
      case 'marcarEnCamino':      result = marcarEnCamino(payload.nro); break;
      case 'guardarOrdenRutaMotorizado': result = guardarOrdenRutaMotorizado(payload); break;
      case 'getPhotoBase64':      result = getPhotoBase64(payload.nro); break;
      default:                    result = { success: false, message: 'Acción no encontrada' };
    }
    return output.setContent(JSON.stringify(result));
  } catch (err) {
    output.setContent(JSON.stringify({ success: false, message: err.toString() }));
  }
  return output;
}

const SPREADSHEET_ID = '1wFzE3tJCLsx-bBoOnLcyCsBngPu75JLfUaFTK_mX62U';

function getSheet() {
  return SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName('Pedidos');
}

function login(user, pass) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName('Usuarios');
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) == user && String(data[i][1]) == pass) {
      return { success: true, user: { usuario: data[i][0], nombre: data[i][2], rol: data[i][3] || 'Lector' } };
    }
  }
  return { success: false, message: "Credenciales incorrectas" };
}

function loginMotorizado(user, pass) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName('Motorizados');
  if (!sheet) return { success: false, message: "Hoja de Motorizados no encontrada" };
  
  const data = sheet.getDataRange().getValues();
  // Asumimos Fila 1 = Encabezados, Fila 2 en adelante = Datos (Col0: Nombre, Col1: Contraseña)
  for (let i = 1; i < data.length; i++) {
    const dbUser = String(data[i][0]).trim();
    const dbPass = String(data[i][1]).trim();
    if (dbUser.toLowerCase() === String(user).trim().toLowerCase() && dbPass === String(pass).trim()) {
      return { success: true, user: dbUser };
    }
  }
  return { success: false, message: "Contraseña incorrecta o usuario no encontrado" };
}

function listarPedidos() {
  const sheet = getSheet();
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return { success: true, data: [] };

  // Leer 19 columnas (A..S)
  const data = sheet.getRange(2, 1, lastRow - 1, 19).getValues();
  // Traemos también los Textos (Display Values) para ignorar el formato histórico de Time/Dates en Sheets
  const displayData = sheet.getRange(2, 1, lastRow - 1, 19).getDisplayValues();

  const pedidos = data.map((row, index) => ({
    nro:               row[0],
    fecha:             row[1],
    llave:             row[2],
    monto:             row[3],
    foto:              row[4],   // E: URL limpia
    monto_foto:        row[5],   // F
    estado:            row[6],   // G
    validado_por:      row[7],   // H
    fecha_validacion:  row[8],   // I
    envio:             row[9],   // J
    pago:              row[10],  // K: Pago Original
    tipo_pago:         row[11],  // L: Tipo Pago Val.
    vuelto:            row[12],  // M: Vuelto
    motivo_cancelacion: row[13], // N: Motivo Cancelación
    sla_fuera:         row[14],  // O: SLA Fuera Tiempo
    fecha_entrega:     displayData[index][15],  // P: Fecha Entrega como texto en hoja
    hora_entrega:      displayData[index][16],  // Q: Hora Entrega como texto en hoja
    tiempo_transcurrido: displayData[index][17], // R: Texto Puro de Tiempo Transcurrido (ej "00:26:00")
    orden_ruta:        row[18]   // S: Ordenamiento explícito para la ruta del motorizado
  })).filter(p => p.nro);

  return { success: true, data: pedidos.sort((a, b) => b.nro - a.nro) };
}

function marcarSLAFuera(nro, fuera_sla) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] == nro) {
      const cell = sheet.getRange(i + 1, 15);
      if (fuera_sla) {
        cell.setValue(new Date()); // Marcar con fecha/hora actual
        return { success: true, marked: true };
      } else {
        cell.clearContent();       // Limpiar correctamente la celda
        return { success: true, marked: false };
      }
    }
  }
  return { success: false, msg: 'Pedido no encontrado para validar.' };
}

function marcarPorValidar(nro, fechaEntrega, horaEntrega) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] == nro) {
      if (rows[i][6] === 'Pendiente' || rows[i][6] === 'En Camino' || rows[i][6] === 'Reservado') {
        sheet.getRange(i + 1, 7).setValue('Por Validar'); // G: Estado = Por Validar
        
        // Si vienen datos de fecha/hora (desde la App del Repartidor al cancelar)
        if (fechaEntrega) sheet.getRange(i + 1, 16).setValue(fechaEntrega); // P
        if (horaEntrega) sheet.getRange(i + 1, 17).setValue(horaEntrega); // Q
        
        return { success: true };
      } else {
         // Si por alguna razón la oficina ya lo validó mientras el patita manejaba
         return { success: true, msg: 'El pedido ya había cambiado de estado' };
      }
    }
  }
  return { success: false, message: "Pedido no encontrado" };
}

function marcarEnCamino(nro) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] == nro) {
      // Solo cambiar si está en Pendiente para no sobreescribir otros estados
      if (rows[i][6] === 'Pendiente') {
        sheet.getRange(i + 1, 7).setValue('En Camino');
        return { success: true };
      }
      return { success: true, message: 'Estado actual no es Pendiente' };
    }
  }
  return { success: false, message: "Pedido no encontrado" };
}

function asignarMotorizado(data) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.nro) {
      sheet.getRange(i + 1, 10).setValue(data.envio || ""); // Columna J: Responsable Envío
      return { success: true };
    }
  }
  return { success: false, message: "Pedido no encontrado" };
}

function validarPedido(data) {
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  var rowIndex = -1;
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.nro) { rowIndex = i + 1; break; }
  }
  if (rowIndex === -1) return { success: false, message: "Pedido no encontrado" };

  var photoUrl = "";
  if (data.archivo && data.archivo.data) {
    try {
      var folderId = "1YO8veJ9RtTGOBdilEEnVHdPSLwsbGdRo";
      var folder = DriveApp.getFolderById(folderId);
      var blob = Utilities.newBlob(
        Utilities.base64Decode(data.archivo.data),
        data.archivo.type,
        data.archivo.name
      );
      var file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      photoUrl = "https://lh3.googleusercontent.com/d/" + file.getId();
    } catch(e) {
      photoUrl = "Error al subir: " + e.toString();
    }
  } else {
    // Mantener URL existente de columna E (ya está limpia)
    photoUrl = rows[rowIndex - 1][4] || "";
    // Si la URL guardada tiene texto concatenado por versiones anteriores, limpiarla
    if (photoUrl && photoUrl.indexOf(' ') > -1) {
      photoUrl = photoUrl.split(' ')[0];
    }
  }

  // Guardar en columnas separadas y limpias
  sheet.getRange(rowIndex, 5).setValue(photoUrl);           // E: URL limpia
  sheet.getRange(rowIndex, 6).setValue(data.montoFoto || 0); // F: Monto foto OCR
  sheet.getRange(rowIndex, 7).setValue("Validado");          // G: Estado
  sheet.getRange(rowIndex, 8).setValue(data.usuario);        // H: Validado por
  sheet.getRange(rowIndex, 9).setValue(new Date());          // I: Fecha validación
  sheet.getRange(rowIndex, 10).setValue(data.envio || "");    // J: Responsable Envío
  sheet.getRange(rowIndex, 12).setValue(data.tipo || "");    // L: Tipo Pago Val.
  sheet.getRange(rowIndex, 13).setValue(                     // M: Vuelto
    (data.tipo === "EFECTIVO" && data.vuelto) ? parseFloat(data.vuelto) : ""
  );
  sheet.getRange(rowIndex, 16).setValue(data.fechaEntrega || ""); // P: Fecha de Entrega
  sheet.getRange(rowIndex, 17).setValue(data.horaEntrega || "");  // Q: Hora de Entrega
  sheet.getRange(rowIndex, 18).setValue(data.tiempoTranscurrido || ""); // R: Tiempo Transcurrido

  return { success: true };
}

// Sube la foto pura a Drive y la asocia al Pedido sin validarlo aún.
function guardarFotoOcrSilent(llaveBusqueda, fotoB64) {
  if (!fotoB64 || !llaveBusqueda) return { success: false, msg: 'Faltan datos' };
  
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  // Busca el pedido por Llave exacta (ignorando vacíos)
  const llaveTarget = llaveBusqueda.toString().trim().toUpperCase();
  let rowIndex = -1;
  
  for (let i = 1; i < data.length; i++) {
    const llaveActual = (data[i][2] || '').toString().trim().toUpperCase(); // Columna C es indice 2 (Llave)
    if (llaveActual === llaveTarget && llaveActual !== '') {
      rowIndex = i + 1;
      break;
    }
  }

  if (rowIndex === -1) {
    return { success: false, msg: 'No se encontró la llave: ' + llaveTarget };
  }

  try {
    const rootFolder = DriveApp.getFolderById('1yXvMbRZ2MAIaNkY0E1VGAr4e1y12L_W_'); // ID de la nueva carpeta FotosApp
    const mimeType = 'image/jpeg';
    const blob = Utilities.newBlob(Utilities.base64Decode(fotoB64), mimeType, 'pre_ocr_' + llaveTarget + '_' + Date.now() + '.jpg');
    const file = rootFolder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    const link = file.getUrl();
    // Guardar el enlace en la Columna E (Foto URL) sin tocar el resto (el estado sigue Pendiente)
    sheet.getRange(rowIndex, 5).setValue(link);
    return { success: true, url: link };
  } catch (e) {
    return { success: false, msg: 'Error de Drive: ' + e.message };
  }
}

function crearPedidosMasivos(payload) {
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  let maxNro = 0;
  const existingKeys = new Set();
  for (let i = 1; i < data.length; i++) {
    const n = parseInt(data[i][0]);
    if (n > maxNro) maxNro = n;
    if (data[i][2]) existingKeys.add(String(data[i][2]).trim());
  }

  payload.orders.forEach(order => {
    if (existingKeys.has(String(order.llave).trim())) return;
    maxNro++;
    sheet.appendRow([
      maxNro,           // A: Nro
      order.fecha,      // B: Fecha
      order.llave,      // C: Llave
      parseFloat(order.monto), // D: Monto
      "",               // E: Foto URL (vacía, se llena al validar)
      0,                // F: Monto Foto
      "Pendiente",      // G: Estado
      "",               // H: Validado Por
      "",               // I: Fecha Validación
      order.envio || "", // J: Responsable Envío
      order.pago || "", // K: Pago Original
      "",               // L: Tipo Pago Val. (se llenará al validar)
      "",               // M: Vuelto
      ""                // N: Motivo Cancelación
    ]);
  });

  return { success: true };
}

function crearPedido(data) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  let maxNro = 0;
  for (let i = 1; i < rows.length; i++) { if (rows[i][0] > maxNro) maxNro = rows[i][0]; }

  sheet.appendRow([
    parseInt(data.nro) || maxNro + 1,
    new Date(),
    data.llave,
    data.monto,
    "", 0, "Pendiente", "", "",
    data.envio || "",
    data.pago || "", "", "", ""
  ]);
  return { success: true };
}

function rechazarPedido(data) {
  const sheet = getSheet();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.nro) {
      sheet.getRange(i + 1, 7).setValue("Cancelado");          // G: Estado
      sheet.getRange(i + 1, 8).setValue(data.usuario);         // H
      sheet.getRange(i + 1, 9).setValue(new Date());           // I
      sheet.getRange(i + 1, 10).setValue(data.envio || "");    // J: Responsable Envío
      sheet.getRange(i + 1, 14).setValue(data.motivo || "");   // N: Motivo Cancelación
      
      // Guardar fecha/hora de evidencia (posiblemente editada por el admin)
      if (data.fechaEntrega) sheet.getRange(i + 1, 16).setValue(data.fechaEntrega); // P
      if (data.horaEntrega) sheet.getRange(i + 1, 17).setValue(data.horaEntrega);   // Q

      // Guardar tiempo transcurrido (R)
      if (data.tiempoTranscurrido) sheet.getRange(i + 1, 18).setValue(data.tiempoTranscurrido);
      
      return { success: true };
    }
  }
  return { success: false, message: "Pedido no encontrado" };
}

function setGeminiKey(data) {
  if (!data.apiKey) return { success: false, message: "API key requerida" };
  PropertiesService.getScriptProperties().setProperty('GEMINI_API_KEY', data.apiKey);
  return { success: true, message: "API key guardada correctamente" };
}

function processVoucherOCR(data) {
  var apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
  if (!apiKey) {
    return { success: false, message: "No hay API key de Gemini configurada. Use setGeminiKey primero." };
  }

  if (!data.imageBase64 || !data.mimeType) {
    return { success: false, message: "Faltan datos de imagen (imageBase64, mimeType)" };
  }

  var prompt = 'Analiza esta imagen que puede ser un voucher POS o una captura de pantalla ONLINE. '
    + 'Extrae estos campos y devuelve SOLO JSON valido sin markdown ni backticks: '
    + '{"fecha":"DD/MM/YY","total":0.00,"tipoPago":"QR o TARJETA","esOnlineValido":false} '
    + 'Reglas: total es monto TOTAL en decimal. tipoPago es QR si menciona QR/Yape/Plin, sino TARJETA. '
    + 'esOnlineValido DEBE ser true SOLO SI se encuentra exactamente el texto "Tarjeta de crédito o débito en línea" en cualquier parte de la imagen, de lo contrario false. '
    + 'Ignora la hora. Si no es visible usa "" o 0. SOLO JSON.';

  // Try models in order
  var models = ['gemini-2.0-flash-lite', 'gemini-2.0-flash'];
  var lastError = '';

  for (var m = 0; m < models.length; m++) {
    var model = models[m];
    var url = 'https://generativelanguage.googleapis.com/v1beta/models/' + model + ':generateContent?key=' + apiKey;

    var body = {
      contents: [{ parts: [
        { text: prompt },
        { inline_data: { mime_type: data.mimeType, data: data.imageBase64 } }
      ]}],
      generationConfig: { temperature: 0.1, maxOutputTokens: 300 }
    };

    try {
      var response = UrlFetchApp.fetch(url, {
        method: 'post',
        contentType: 'application/json',
        payload: JSON.stringify(body),
        muteHttpExceptions: true
      });

      var code = response.getResponseCode();
      var respText = response.getContentText();

      if (code === 429) {
        lastError = 'Cuota excedida para ' + model;
        Utilities.sleep(2000);
        continue;
      }

      if (code !== 200) {
        lastError = 'Error ' + code + ' en ' + model + ': ' + respText;
        continue;
      }

      var jsonResp = JSON.parse(respText);
      var geminiText = jsonResp.candidates[0].content.parts[0].text || '';
      
      // Clean markdown wrapping if present
      geminiText = geminiText.replace(/^```json?\s*/i, '').replace(/```\s*$/i, '').trim();
      
      // Extract JSON object
      var jsonMatch = geminiText.match(/\{[\s\S]*\}/);
      if (jsonMatch) {
        var result = JSON.parse(jsonMatch[0]);
        result.model = model;
        return { success: true, data: result };
      } else {
        lastError = 'Respuesta no parseable: ' + geminiText.substring(0, 100);
      }
    } catch(e) {
      lastError = e.toString();
    }
  }

  return { success: false, message: 'OCR falló: ' + lastError };
}

function guardarOrdenRutaMotorizado(payload) {
  try {
    const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName('Pedidos');
    if (!sheet) return { success: false, message: 'Hoja Pedidos no encontrada' };

    const data = sheet.getDataRange().getValues();
    const orderedIds = payload.orderedIds || [];
    const responsable = payload.responsable;

    if (!responsable) return { success: false, message: 'Falta nombre del responsable' };

    let updatesCount = 0;

    for (let i = 1; i < data.length; i++) {
       const rowNro = data[i][0].toString();
       const rowResp = (data[i][9] || '').toString().trim(); // Col J 
       
       if (rowResp.toLowerCase() === responsable.toLowerCase()) {
         // Find this NRO in the orderedIds array
         const orderIndex = orderedIds.indexOf(rowNro);
         
         if (orderIndex !== -1) {
            // Found it, set its route order (Index 0, 1, 2...)
            // Guardamos en la columna S (Index 18 original es 0-based, por tanto Columna 19 es 1-based getRange)
            // A=1, B=2 ... R=18, S=19. Si la columna "Orden Ruta" es la S, entonces es 19. 
            sheet.getRange(i + 1, 19).setValue(orderIndex);
            updatesCount++;
         }
       }
    }

    return { success: true, message: `Rutas guardadas: ${updatesCount}` };
  } catch(e) {
    return { success: false, message: e.toString() };
  }
}
function getPhotoBase64(nro) {
  try {
    const sheet = getSheet();
    const rows = sheet.getDataRange().getValues();
    let url = "";
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] == nro) {
        url = (rows[i][4] || "").toString().trim();
        break;
      }
    }
    
    if (!url) return { success: false, message: "No hay URL de foto para este pedido" };
    
    // Extraer ID de Drive
    let id = "";
    const idMatch = url.match(/\/d\/([a-zA-Z0-9_-]+)/) || url.match(/[?&]id=([a-zA-Z0-9_-]+)/);
    if (idMatch) {
      id = idMatch[1];
    } else if (url.length > 20 && !url.includes('/') && !url.includes('.')) {
      id = url;
    } else {
      return { success: false, message: "No se pudo extraer el ID de Drive de la URL" };
    }
    
    const file = DriveApp.getFileById(id);
    const blob = file.getBlob();
    const b64 = Utilities.base64Encode(blob.getBytes());
    
    return { success: true, base64: b64, mimeType: blob.getContentType() };
  } catch(e) {
    return { success: false, message: e.toString() };
  }
}
